# Admin class management — implementation plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship admin CRUD for school classes — add, rename, reassign the class teacher, delete — under `Admin → People → Classes`, in one PR (`feat/admin-class-management`).

**Architecture:** Reuse the `holidays` module pattern (repo → service → routes, admin-gated). One Drizzle schema change adds a FK + unique index on `classes.teacher_id`. `classSchema` gains an optional server-aggregated `studentCount`. A new lightweight `users` module exposes `GET /users?role=teacher` for the picker. Frontend is `AdminCalendarPage`-shaped: one page, three modals (add/edit/delete) sharing a small form. No new dependencies. No cron, notifications, or enum changes.

**Tech Stack:** Drizzle ORM + Postgres, Fastify v5, Zod via `fastify-type-provider-zod`, Vitest + supertest-style `app.inject`, React 18 + TanStack Query v5, MSW v2, react-i18next.

**Spec:** `docs/superpowers/specs/2026-05-17-admin-class-management-design.md`.

**Branch:** `feat/admin-class-management` (already created, off latest `main`).

Append the methods below to the `classesRepo` object literal in `apps/api/src/modules/classes/repository.ts`. Add `ilike` to the drizzle import list (`import { ..., ilike } from 'drizzle-orm'`). Add `users` to the imports (`import { users } from '../../db/schema/auth.js'`).

```ts
  // --- Admin CRUD helpers ------------------------------------------------

  async create(
    ctx: TenantContext,
    input: { name: string; teacherId: string },
  ) {
    const id = newId()
    const rows = await db
      .insert(classes)
      .values({
        id,
        schoolId: ctx.schoolId,
        name: input.name,
        teacherId: input.teacherId,
      })
      .returning()
    return rows[0]!
  },

  async patch(
    ctx: TenantContext,
    id: string,
    input: { name?: string; teacherId?: string },
  ) {
    const patch: Partial<Pick<typeof classes.$inferSelect, 'name' | 'teacherId' | 'updatedAt'>> = {
      updatedAt: new Date(),
    }
    if (input.name !== undefined) patch.name = input.name
    if (input.teacherId !== undefined) patch.teacherId = input.teacherId
    const rows = await db
      .update(classes)
      .set(patch)
      .where(and(eq(classes.schoolId, ctx.schoolId), eq(classes.id, id)))
      .returning()
    return rows[0]
  },

  async delete(ctx: TenantContext, id: string): Promise<boolean> {
    const rows = await db
      .delete(classes)
      .where(and(eq(classes.schoolId, ctx.schoolId), eq(classes.id, id)))
      .returning({ id: classes.id })
    return rows.length > 0
  },

  async countStudents(ctx: TenantContext, classId: string): Promise<number> {
    const rows = await db
      .select({ n: count() })
      .from(students)
      .where(
        and(
          eq(students.schoolId, ctx.schoolId),
          eq(students.classId, classId),
        ),
      )
    return rows[0]?.n ?? 0
  },

  async findByTeacher(ctx: TenantContext, teacherId: string) {
    const rows = await db
      .select()
      .from(classes)
      .where(and(eq(classes.schoolId, ctx.schoolId), eq(classes.teacherId, teacherId)))
      .limit(1)
    return rows[0]
  },

  async findByNameCaseInsensitive(ctx: TenantContext, name: string) {
    const rows = await db
      .select()
      .from(classes)
      .where(and(eq(classes.schoolId, ctx.schoolId), ilike(classes.name, name)))
      .limit(1)
    return rows[0]
  },

  async findTeacherById(ctx: TenantContext, userId: string) {
    const rows = await db
      .select({ id: users.id, role: users.role })
      .from(users)
      .where(and(eq(users.schoolId, ctx.schoolId), eq(users.id, userId)))
      .limit(1)
    return rows[0]
  },
```

```ts
    throw new ConflictError(`A class named "${name}" already exists`, 'CLASS_NAME_TAKEN')
  }
}

async function assertTeacherAvailable(
  ctx: TenantContext,
  teacherId: string,
  excludingClassId?: string,
) {
  const existing = await classesRepo.findByTeacher(ctx, teacherId)
  if (existing && existing.id !== excludingClassId) {
    throw new ConflictError(
      `Teacher is already assigned to another class`,
      'TEACHER_ALREADY_ASSIGNED',
    )
  }
}

export async function createClass(
  ctx: TenantContext,
  input: CreateClassRequest,
): Promise<Class> {
  if (ctx.role !== 'admin') throw new ForbiddenError()
  await assertEligibleTeacher(ctx, input.teacherId)
  await assertNameAvailable(ctx, input.name)
  await assertTeacherAvailable(ctx, input.teacherId)
  try {
    const row = await classesRepo.create(ctx, input)
    return classToWire(row, 0)
  } catch (err) {
    // DB unique-constraint backstop: classes_school_teacher_unique
    if (err instanceof Error && /classes_school_teacher_unique/.test(err.message)) {
      throw new ConflictError(
        'Teacher is already assigned to another class',
        'TEACHER_ALREADY_ASSIGNED',
      )
    }
    throw err
  }
}
```

```ts
      void client.invalidateQueries({ queryKey: ['users', 'teachers'] })
    },
  })
}

export function useDeleteClass() {
  const client = useQueryClient()
  return useMutation<{ ok: true }, Error, string>({
    mutationFn: (id) => apiDelete(`/classes/${id}`, okResponseSchema),
    onSuccess: () => {
      void client.invalidateQueries({ queryKey: classKeys.list })
      void client.invalidateQueries({ queryKey: ['users', 'teachers'] })
    },
  })
}
```

```tsx
            <table className="min-w-full divide-y divide-stone-200 text-sm">

                <tr>
```

```ts
import { http, HttpResponse } from 'msw'
import { seed } from './seed'

// ...existing handlers...

// GET /classes — surface studentCount alongside each class.
http.get('/api/classes', () => {
  const list = seed.classes.map((c) => ({
    ...c,
    studentCount: seed.students.filter((s) => s.classId === c.id).length,
  }))
  return HttpResponse.json(list)
}),

http.post('/api/classes', async ({ request }) => {
  const body = (await request.json()) as { name: string; teacherId: string }
  // Uniqueness checks mirrored from the backend.
  if (
    seed.classes.some(
      (c) => c.name.toLowerCase() === body.name.trim().toLowerCase(),
    )
  ) {
    return HttpResponse.json(
      { statusCode: 409, error: 'ConflictError', message: 'name taken', code: 'CLASS_NAME_TAKEN' },
      { status: 409 },
    )
  }
  if (seed.classes.some((c) => c.teacherId === body.teacherId)) {
    return HttpResponse.json(
      { statusCode: 409, error: 'ConflictError', message: 'teacher taken', code: 'TEACHER_ALREADY_ASSIGNED' },
      { status: 409 },
    )
  }
  const teacher = seed.users.find((u) => u.id === body.teacherId && u.role === 'teacher')
  if (!teacher) {
    return HttpResponse.json(
      { statusCode: 400, error: 'ValidationError', message: 'invalid teacher', code: 'TEACHER_NOT_ELIGIBLE' },
      { status: 400 },
    )
  }
  const created = {
    id: crypto.randomUUID(),
    schoolId: seed.school.id,
    name: body.name.trim(),
    teacherId: body.teacherId,
    studentCount: 0,
  }
  seed.classes.push(created)
  return HttpResponse.json(created)
}),

http.patch('/api/classes/:id', async ({ params, request }) => {
  const id = params.id as string
  const body = (await request.json()) as { name?: string; teacherId?: string }
  const target = seed.classes.find((c) => c.id === id)
  if (!target) return HttpResponse.json({}, { status: 404 })

  if (
    body.name &&
    seed.classes.some(
      (c) => c.id !== id && c.name.toLowerCase() === body.name!.trim().toLowerCase(),
    )
  ) {
    return HttpResponse.json(
      { statusCode: 409, error: 'ConflictError', message: 'name taken', code: 'CLASS_NAME_TAKEN' },
      { status: 409 },
    )
  }
  if (
    body.teacherId &&
    seed.classes.some((c) => c.id !== id && c.teacherId === body.teacherId)
  ) {
    return HttpResponse.json(
      { statusCode: 409, error: 'ConflictError', message: 'teacher taken', code: 'TEACHER_ALREADY_ASSIGNED' },
      { status: 409 },
    )
  }
  if (body.name !== undefined) target.name = body.name.trim()
  if (body.teacherId !== undefined) target.teacherId = body.teacherId
  return HttpResponse.json({
    ...target,
    studentCount: seed.students.filter((s) => s.classId === target.id).length,
  })
}),

http.delete('/api/classes/:id', ({ params }) => {
  const id = params.id as string
  const idx = seed.classes.findIndex((c) => c.id === id)
  if (idx < 0) return HttpResponse.json({}, { status: 404 })
  const count = seed.students.filter((s) => s.classId === id).length
  if (count > 0) {
    return HttpResponse.json(
      { statusCode: 409, error: 'ConflictError', message: `has ${count}`, code: 'CLASS_HAS_STUDENTS' },
      { status: 409 },
    )
  }
  seed.classes.splice(idx, 1)
  return HttpResponse.json({ ok: true })
}),

http.get('/api/users', ({ request }) => {
  const url = new URL(request.url)
  if (url.searchParams.get('role') !== 'teacher') {
    return HttpResponse.json({}, { status: 400 })
  }
  return HttpResponse.json(
    seed.users
      .filter((u) => u.role === 'teacher')
      .sort((a, b) => a.fullName.localeCompare(b.fullName))
      .map((u) => ({ id: u.id, fullName: u.fullName })),
  )
}),
```
